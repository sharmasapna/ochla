CLEA
CLOS ALL
USE MS2
DO OCHNFS
CLOS ALL


SELE 1
USE MS2
SELE 2
USE OCHNF

SELE 1
DO WHILE .NOT.EOF()

DO WHILE CL#" "
SKIP
ENDDO

IF STATION=SPACE(18).OR.LEVEL=SPACE(8)
X=RECNO()
?RECNO()
NFS=NFSC
NM="                  "
@10,15 SAY "FOLLOWING NFSC IS TO ENTERED FOR STATION,LEVEL      "+NFS
@12,15 SAY "ENTER STATION NAME" GET NM
READ
@13,15 SAY "CHOOSE LEVEL "
@13,40 PROMP "I       "
@14,40 PROMP "II M.P. "
@15,40 PROMP "II OM.P."
@16,40 PROMP "LOCAL  "
@17,40 PROMP "PRNTD   "
@18,40 PROMP "INCD   "
@19,40 PROMP "PRIVATE"
@20,40 PROMP "BHEL"
MENU TO NF

SELE 2
LOCA FOR NFSC=NFS

IF .NOT.FOUND()
APPE BLAN
ENDIF
REPL NFSC WITH NFS,STATION WITH NM
IF NF=1
REPL LEVEL WITH "I"
ENDIF

IF NF=2
REPL LEVEL WITH "II M.P."
ENDIF

IF NF=3
REPL LEVEL WITH "II OM.P."
ENDIF

IF NF=4
REPL LEVEL WITH "LOCAL"
ENDIF

IF NF=5
REPL LEVEL WITH "PRNTD"
ENDIF

IF NF=6
REPL LEVEL WITH "INCD"
ENDIF

IF NF=7
REPL LEVEL WITH "PRIVATE"
ENDIF

IF NF=8
REPL LEVEL WITH "BHEL"
ENDIF
ENDIF
SELE 1
IF.NOT.EOF()
SKIP
ELSE
EXIT
ENDIF
ENDDO

CLOS ALL
USE OCHNF

REPL ALL A WITH 'REPL ALL STATION WITH "'+ALLTRIM(STATION)+'",LEVEL WITH "'+ALLTRIM(LEVEL)+'" FOR NFSC$"'+NFSC+'"'
COPY FIELD A TO OCHNFS.PRG SDF
SELE 1
DO OCHNFS
IF.NOT.EOF()
SKIP
ELSE

EXIT
ENDIF
ENDDO

CLOS ALL
